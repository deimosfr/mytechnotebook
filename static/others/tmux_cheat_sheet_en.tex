All listed functions listed below require Ctrl+b keyboard press key before

\section{Commands}
\cm{tmux}{Launch tmux}
\cm{tmux a}{Session attach}

\section{Window Managment}
\cm{c}{Create a new window}
\cm{w}{Get current window list}
\cm{n}{Move to next window}
\cm{p}{Move to previous window}
\cm{l}{Move to the penultimate used window}
\cm{[0-9]}{Move to the numbered window}
\cm{f \textit{name}}{Search in window's buffers}
\cm{'}{Rename current window}
\cm{\&}{Force a window to close}
\cm{t}{Show time}
\columnbreak

\section{Split}
\cm{"}{Vertical screen cut}
\cm{\%}{Horizontal screen cut}
\cm{\{}{Move to previous pane}
\cm{\} \textit{or} o}{Move to next pane}
\cm{$\leftarrow$ $\rightarrow$ $\uparrow$ $\downarrow$}{Move to the corresponding arrow}
\cm{q}{Get panes numbers}
\cm{$\sqcup$}{Change panes visual organisation}
\cm{Alt+($\leftarrow$ $\rightarrow$ $\uparrow$ $\downarrow$)}{Grow pane size}
\cm{!}{Convert a pane to a window}
\cm{:joinp -h -s 0.0 -p 75}{Convert a window in an integrated pane}
$\bullet$ -h : horizontaly \\
$\bullet$ -s 0.0 : window 0 et pane 0\\
$\bullet$ -p 75 : 75\% window occupation

\section{History}
\cm{$\Uparrow$ (PageUP)}{View upward in history}
\cm{$\Downarrow$ (PageDOWN)}{View downward in history}
\cm{$\sqcup$ \textit{then}($\uparrow$ $\downarrow$)}{Select lines in history (after $\Uparrow$)}
\cm{$\hookleftarrow$}{Copy selection}
\cm{=}{Paste selection}
\columnbreak

\section{Session}
\cm{d}{Detach tmux session}
\cm{s}{List tmux session}
\cm{(}{Jump to next session}
\cm{)}{Jump to previous sessions}
